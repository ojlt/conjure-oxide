18pub fn parse_essence_file_native(
19    filepath: &str,
20    context: Arc<RwLock<Context<'static>>>,
21) -> Result<Model, EssenceParseError> {
22    let (tree, source_code) = get_tree(filepath);
23
24    let mut model = Model::new(context);
25    let root_node = tree.root_node();
26    for statement in named_children(&root_node) {
27        match statement.kind() {
28            "single_line_comment" => {}
29            "find_statement_list" => {
30                let var_hashmap = parse_find_statement(statement, &source_code);
31                for (name, decision_variable) in var_hashmap {
32                    model
33                        .as_submodel_mut()
34                        .symbols_mut()
35                        .insert(Rc::new(Declaration::new_var(name, decision_variable)));
36                }
37            }
38            "constraint_list" => {
39                let mut constraint_vec: Vec<Expression> = Vec::new();
40                for constraint in named_children(&statement) {
41                    if constraint.kind() != "single_line_comment" {
42                        constraint_vec.push(parse_constraint(constraint, &source_code, &statement));
43                    }
44                }
45                model.as_submodel_mut().add_constraints(constraint_vec);
46            }
47            "language_label" => {}
48            "letting_statement_list" => {
49                let letting_vars = parse_letting_statement(statement, &source_code);
50                model.as_submodel_mut().symbols_mut().extend(letting_vars);
51            }
52            "dominance_relation" => {
53                let inner = statement
54                    .child(1)
55                    .expect("Expected a sub-expression inside `dominanceRelation`");
56                let expr = parse_constraint(inner, &source_code, &statement);
57                let dominance = Expression::DominanceRelation(Metadata::new(), Box::new(expr));
58                if model.dominance.is_some() {
59                    return Err(EssenceParseError::ParseError(Error::Parse(
60                        "Duplicate dominance relation".to_owned(),
61                    )));
62                }
63                model.dominance = Some(dominance);
64            }
65            _ => {
66                let kind = statement.kind();
67                return Err(EssenceParseError::ParseError(Error::Parse(
68                    format!("Unrecognized top level statement kind: {kind}").to_owned(),
69                )));
70            }
71        }
72    }
73    Ok(model)
74}

76fn get_tree(filepath: &str) -> (Tree, String) {
77    let source_code = fs::read_to_string(filepath)
78        .unwrap_or_else(|_| panic!("Failed to read the source code file {}", filepath));
79    let mut parser = Parser::new();
80    parser.set_language(&LANGUAGE.into()).unwrap();
81    (
82        parser
83            .parse(source_code.clone(), None)
84            .expect("Failed to parse"),
85        source_code,
86    )
87}

89fn parse_find_statement(find_statement_list: Node, source_code: &str) -> BTreeMap<Name, Domain> {
90    let mut vars = BTreeMap::new();
91
92    for find_statement in named_children(&find_statement_list) {
93        let mut temp_symbols = BTreeSet::new();
94
95        let variable_list = find_statement
96            .named_child(0)
97            .expect("No variable list found");
98        for variable in named_children(&variable_list) {
99            let variable_name = &source_code[variable.start_byte()..variable.end_byte()];
100            temp_symbols.insert(variable_name);
101        }
102
103        let domain = find_statement.named_child(1).expect("No domain found");
104        let domain = parse_domain(domain, source_code);
105
106        for name in temp_symbols {
107            vars.insert(Name::UserName(String::from(name)), domain.clone());
108        }
109    }
110    vars
111}

113fn parse_domain(domain: Node, source_code: &str) -> Domain {
114    let domain = domain.child(0).expect("No domain");
115    match domain.kind() {
116        "bool_domain" => Domain::BoolDomain,
117        "int_domain" => parse_int_domain(domain, source_code),
118        "variable" => {
119            let variable_name = &source_code[domain.start_byte()..domain.end_byte()];
120            Domain::DomainReference(Name::UserName(String::from(variable_name)))
121        }
122        _ => panic!("Not bool or int domain"),
123    }
124}

126fn parse_int_domain(int_domain: Node, source_code: &str) -> Domain {
127    if int_domain.child_count() == 1 {
128        Domain::IntDomain(vec![Range::Bounded(std::i32::MIN, std::i32::MAX)])
129    } else {
130        let mut ranges: Vec<Range<i32>> = Vec::new();
131        let range_list = int_domain
132            .named_child(0)
133            .expect("No range list found (expression ranges not supported yet");
134        for int_range in named_children(&range_list) {
135            match int_range.kind() {
136                "integer" => {
137                    let integer_value = &source_code[int_range.start_byte()..int_range.end_byte()]
138                        .parse::<i32>()
139                        .unwrap();
140                    ranges.push(Range::Single(*integer_value));
141                }
142                "int_range" => {
143                    let lower_bound: Option<i32>;
144                    let upper_bound: Option<i32>;
145                    let range_component = int_range.child(0).expect("Error with integer range");
146                    match range_component.kind() {
147                        "expression" => {
148                            lower_bound = Some(
149                                source_code
150                                    [range_component.start_byte()..range_component.end_byte()]
151                                    .parse::<i32>()
152                                    .unwrap(),
153                            );
154
155                            if let Some(range_component) = range_component.next_named_sibling() {
156                                upper_bound = Some(
157                                    source_code
158                                        [range_component.start_byte()..range_component.end_byte()]
159                                        .parse::<i32>()
160                                        .unwrap(),
161                                );
162                            } else {
163                                upper_bound = None;
164                            }
165                        }
166                        ".." => {
167                            lower_bound = None;
168                            let range_component = range_component
169                                .next_sibling()
170                                .expect("Error with integer range");
171                            upper_bound = Some(
172                                source_code
173                                    [range_component.start_byte()..range_component.end_byte()]
174                                    .parse::<i32>()
175                                    .unwrap(),
176                            );
177                        }
178                        _ => panic!("unsupported int range type"),
179                    }
180
181                    match (lower_bound, upper_bound) {
182                        (Some(lb), Some(ub)) => ranges.push(Range::Bounded(lb, ub)),
183                        (Some(lb), None) => ranges.push(Range::Bounded(lb, std::i32::MAX)),
184                        (None, Some(ub)) => ranges.push(Range::Bounded(std::i32::MIN, ub)),
185                        _ => panic!("Unsupported int range type"),
186                    }
187                }
188                _ => panic!("unsupported int range type"),
189            }
190        }
191        Domain::IntDomain(ranges)
192    }
193}

195fn parse_letting_statement(letting_statement_list: Node, source_code: &str) -> SymbolTable {
196    let mut symbol_table = SymbolTable::new();
197
198    for letting_statement in named_children(&letting_statement_list) {
199        let mut temp_symbols = BTreeSet::new();
200
201        let variable_list = letting_statement.child(0).expect("No variable list found");
202        for variable in named_children(&variable_list) {
203            let variable_name = &source_code[variable.start_byte()..variable.end_byte()];
204            temp_symbols.insert(variable_name);
205        }
206
207        let expr_or_domain = letting_statement
208            .named_child(1)
209            .expect("No domain or expression found for letting statement");
210        match expr_or_domain.kind() {
211            "expression" => {
212                for name in temp_symbols {
213                    symbol_table.insert(Rc::new(Declaration::new_value_letting(
214                        Name::UserName(String::from(name)),
215                        parse_constraint(expr_or_domain, source_code, &letting_statement_list),
216                    )));
217                }
218            }
219            "domain" => {
220                for name in temp_symbols {
221                    symbol_table.insert(Rc::new(Declaration::new_domain_letting(
222                        Name::UserName(String::from(name)),
223                        parse_domain(expr_or_domain, source_code),
224                    )));
225                }
226            }
227            _ => panic!("Unrecognized node in letting statement"),
228        }
229    }
230    symbol_table
231}

233fn parse_constraint(constraint: Node, source_code: &str, root: &Node) -> Expression {
234    match constraint.kind() {
235        "constraint" | "expression" | "boolean_expr" | "comparison_expr" | "arithmetic_expr"
236        | "primary_expr" | "sub_expr" => child_expr(constraint, source_code, root),
237        "not_expr" => Expression::Not(
238            Metadata::new(),
239            Box::new(child_expr(constraint, source_code, root)),
240        ),
241        "abs_value" => Expression::Abs(
242            Metadata::new(),
243            Box::new(child_expr(constraint, source_code, root)),
244        ),
245        "negative_expr" => Expression::Neg(
246            Metadata::new(),
247            Box::new(child_expr(constraint, source_code, root)),
248        ),
249        "exponent" | "product_expr" | "sum_expr" | "comparison" | "and_expr" | "or_expr"
250        | "implication" => {
251            let expr1 = child_expr(constraint, source_code, root);
252            let op = constraint.child(1).unwrap_or_else(|| {
253                panic!(
254                    "Error: missing node in expression of kind {}",
255                    constraint.kind()
256                )
257            });
258            let op_type = &source_code[op.start_byte()..op.end_byte()];
259            let expr2_node = constraint.child(2).unwrap_or_else(|| {
260                panic!(
261                    "Error: missing node in expression of kind {}",
262                    constraint.kind()
263                )
264            });
265            let expr2 = parse_constraint(expr2_node, source_code, root);
266
267            match op_type {
268                "**" => Expression::UnsafePow(Metadata::new(), Box::new(expr1), Box::new(expr2)),
269                "+" => Expression::Sum(Metadata::new(), Box::new(matrix_expr![expr1, expr2])),
270                "-" => Expression::Minus(Metadata::new(), Box::new(expr1), Box::new(expr2)),
271                "*" => Expression::Product(Metadata::new(), vec![expr1, expr2]),
272                "/" => {
273                    //TODO: add checks for if division is safe or not
274                    Expression::UnsafeDiv(Metadata::new(), Box::new(expr1), Box::new(expr2))
275                }
276                "%" => {
277                    //TODO: add checks for if mod is safe or not
278                    Expression::UnsafeMod(Metadata::new(), Box::new(expr1), Box::new(expr2))
279                }
280                "=" => Expression::Eq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
281                "!=" => Expression::Neq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
282                "<=" => Expression::Leq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
283                ">=" => Expression::Geq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
284                "<" => Expression::Lt(Metadata::new(), Box::new(expr1), Box::new(expr2)),
285                ">" => Expression::Gt(Metadata::new(), Box::new(expr1), Box::new(expr2)),
286                "/\\" => Expression::And(Metadata::new(), Box::new(matrix_expr![expr1, expr2])),
287                "\\/" => Expression::Or(Metadata::new(), Box::new(matrix_expr![expr1, expr2])),
288                "->" => Expression::Imply(Metadata::new(), Box::new(expr1), Box::new(expr2)),
289                _ => panic!("Error: unsupported operator"),
290            }
291        }
292        "quantifier_expr" => {
293            let mut expr_list = Vec::new();
294            for expr in named_children(&constraint) {
295                expr_list.push(parse_constraint(expr, source_code, root));
296            }
297
298            let quantifier = constraint.child(0).unwrap_or_else(|| {
299                panic!(
300                    "Error: missing node in expression of kind {}",
301                    constraint.kind()
302                )
303            });
304            let quantifier_type = &source_code[quantifier.start_byte()..quantifier.end_byte()];
305
306            match quantifier_type {
307                "and" => Expression::And(Metadata::new(), Box::new(into_matrix_expr![expr_list])),
308                "or" => Expression::Or(Metadata::new(), Box::new(into_matrix_expr![expr_list])),
309                "min" => Expression::Min(Metadata::new(), Box::new(into_matrix_expr![expr_list])),
310                "max" => Expression::Max(Metadata::new(), Box::new(into_matrix_expr![expr_list])),
311                "sum" => Expression::Sum(Metadata::new(), Box::new(into_matrix_expr![expr_list])),
312                "allDiff" => {
313                    Expression::AllDiff(Metadata::new(), Box::new(into_matrix_expr![expr_list]))
314                }
315                _ => panic!("Error: unsupported quantifier"),
316            }
317        }
318        "constant" => {
319            let child = constraint.child(0).unwrap_or_else(|| {
320                panic!(
321                    "Error: missing node in expression of kind {}",
322                    constraint.kind()
323                )
324            });
325            match child.kind() {
326                "integer" => {
327                    let constant_value = &source_code[child.start_byte()..child.end_byte()]
328                        .parse::<i32>()
329                        .unwrap();
330                    Expression::Atomic(
331                        Metadata::new(),
332                        Atom::Literal(Literal::Int(*constant_value)),
333                    )
334                }
335                "TRUE" => Expression::Atomic(Metadata::new(), Atom::Literal(Literal::Bool(true))),
336                "FALSE" => Expression::Atomic(Metadata::new(), Atom::Literal(Literal::Bool(false))),
337                _ => panic!("Error"),
338            }
339        }
340        "variable" => {
341            let variable_name =
342                String::from(&source_code[constraint.start_byte()..constraint.end_byte()]);
343            Expression::Atomic(
344                Metadata::new(),
345                Atom::Reference(Name::UserName(variable_name)),
346            )
347        }
348        "ERROR" => {
349            panic!("");
350        }
351        "from_solution" => match root.kind() {
352            "dominance_relation" => {
353                let inner = child_expr(constraint, source_code, root);
354                match inner {
355                    Expression::Atomic(_, _) => {
356                        Expression::FromSolution(Metadata::new(), Box::new(inner))
357                    }
358                    _ => panic!("Expression inside a `fromSolution()` must be a variable name"),
359                }
360            }
361            _ => panic!("`fromSolution()` is only allowed inside dominance relation definitions"),
362        },
363        _ => panic!("{} is not a recognized node kind", constraint.kind()),
364    }
365}

367fn named_children<'a>(node: &'a Node<'a>) -> impl Iterator<Item = Node<'a>> + 'a {
368    (0..node.named_child_count()).filter_map(|i| node.named_child(i))
369}

371fn child_expr(node: Node, source_code: &str, root: &Node) -> Expression {
372    let child = node
373        .named_child(0)
374        .unwrap_or_else(|| panic!("Error: missing node in expression of kind {}", node.kind()));
375    parse_constraint(child, source_code, root)
