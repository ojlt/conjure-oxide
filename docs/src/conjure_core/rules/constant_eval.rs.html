18#[register_rule(("Constant", 9001))]
19fn apply_eval_constant(expr: &Expr, _: &SymbolTable) -> ApplicationResult {
20    if let Expr::Atomic(_, Atom::Literal(_)) = expr {
21        return Err(ApplicationError::RuleNotApplicable);
22    }
23    eval_constant(expr)
24        .map(|c| Reduction::pure(Expr::Atomic(Metadata::new(), Atom::Literal(c))))
25        .ok_or(ApplicationError::RuleNotApplicable)
26}

28/// Simplify an expression to a constant if possible
29/// Returns:
30/// `None` if the expression cannot be simplified to a constant (e.g. if it contains a variable)
31/// `Some(Const)` if the expression can be simplified to a constant
32pub fn eval_constant(expr: &Expr) -> Option<Lit> {
33    match expr {
34        Expr::FromSolution(_, _) => None,
35        Expr::DominanceRelation(_, _) => None,
36        Expr::InDomain(_, e, domain) => {
37            let Expr::Atomic(_, Atom::Literal(lit)) = e.as_ref() else {
38                return None;
39            };
40
41            domain.contains(lit).map(Into::into)
42        }
43        Expr::Atomic(_, Atom::Literal(c)) => Some(c.clone()),
44        Expr::Atomic(_, Atom::Reference(_c)) => None,
45        Expr::AbstractLiteral(_, _) => None,
46        Expr::Comprehension(_, _) => None,
47        Expr::UnsafeIndex(_, subject, indices) | Expr::SafeIndex(_, subject, indices) => {
48            let subject: Lit = subject.as_ref().clone().to_literal()?;
49            let indices: Vec<Lit> = indices
50                .iter()
51                .cloned()
52                .map(|x| x.to_literal())
53                .collect::<Option<Vec<Lit>>>()?;
54
55            let Lit::AbstractLiteral(subject @ AbstractLiteral::Matrix(_, _)) = subject else {
56                return None;
57            };
58
59            matrix::flatten_enumerate(subject)
60                .find(|(i, _)| i == &indices)
61                .map(|(_, x)| x)
62        }
63        Expr::UnsafeSlice(_, subject, indices) | Expr::SafeSlice(_, subject, indices) => {
64            let subject: Lit = subject.as_ref().clone().to_literal()?;
65            let Lit::AbstractLiteral(subject @ AbstractLiteral::Matrix(_, _)) = subject else {
66                return None;
67            };
68
69            let hole_dim = indices
70                .iter()
71                .cloned()
72                .position(|x| x.is_none())
73                .expect("slice expression should have a hole dimension");
74
75            let missing_domain = matrix::index_domains(subject.clone())[hole_dim].clone();
76
77            let indices: Vec<Option<Lit>> = indices
78                .iter()
79                .cloned()
80                .map(|x| {
81                    // the outer option represents success of this iterator, the inner the index
82                    // slice.
83                    match x {
84                        Some(x) => x.to_literal().map(Some),
85                        None => Some(None),
86                    }
87                })
88                .collect::<Option<Vec<Option<Lit>>>>()?;
89
90            let indices_in_slice: Vec<Vec<Lit>> = missing_domain
91                .values()?
92                .into_iter()
93                .map(|i| {
94                    let mut indices = indices.clone();
95                    indices[hole_dim] = Some(i);
96                    // These unwraps will only fail if we have multiple holes.
97                    // As this is invalid, panicking is fine.
98                    indices.into_iter().map(|x| x.unwrap()).collect_vec()
99                })
100                .collect_vec();
101
102            // Note: indices_in_slice is not necessarily sorted, so this is the best way.
103            let elems = matrix::flatten_enumerate(subject)
104                .filter(|(i, _)| indices_in_slice.contains(i))
105                .map(|(_, elem)| elem)
106                .collect();
107
108            Some(Lit::AbstractLiteral(into_matrix![elems]))
109        }
110        Expr::Abs(_, e) => un_op::<i32, i32>(|a| a.abs(), e).map(Lit::Int),
111        Expr::Eq(_, a, b) => bin_op::<i32, bool>(|a, b| a == b, a, b)
112            .or_else(|| bin_op::<bool, bool>(|a, b| a == b, a, b))
113            .map(Lit::Bool),
114        Expr::Neq(_, a, b) => bin_op::<i32, bool>(|a, b| a != b, a, b).map(Lit::Bool),
115        Expr::Lt(_, a, b) => bin_op::<i32, bool>(|a, b| a < b, a, b).map(Lit::Bool),
116        Expr::Gt(_, a, b) => bin_op::<i32, bool>(|a, b| a > b, a, b).map(Lit::Bool),
117        Expr::Leq(_, a, b) => bin_op::<i32, bool>(|a, b| a <= b, a, b).map(Lit::Bool),
118        Expr::Geq(_, a, b) => bin_op::<i32, bool>(|a, b| a >= b, a, b).map(Lit::Bool),
119        Expr::Not(_, expr) => un_op::<bool, bool>(|e| !e, expr).map(Lit::Bool),
120        Expr::And(_, e) => {
121            vec_lit_op::<bool, bool>(|e| e.iter().all(|&e| e), e.as_ref()).map(Lit::Bool)
122        }
123        Expr::Root(_, _) => None,
124        Expr::Or(_, e) => {
125            vec_lit_op::<bool, bool>(|e| e.iter().any(|&e| e), e.as_ref()).map(Lit::Bool)
126        }
127        Expr::Imply(_, box1, box2) => {
128            let a: &Atom = (&**box1).try_into().ok()?;
129            let b: &Atom = (&**box2).try_into().ok()?;
130
131            let a: bool = a.try_into().ok()?;
132            let b: bool = b.try_into().ok()?;
133
134            if a {
135                // true -> b ~> b
136                Some(Lit::Bool(b))
137            } else {
138                // false -> b ~> true
139                Some(Lit::Bool(true))
140            }
141        }
142        Expr::Sum(_, exprs) => vec_lit_op::<i32, i32>(|e| e.iter().sum(), exprs).map(Lit::Int),
143        Expr::Product(_, exprs) => vec_op::<i32, i32>(|e| e.iter().product(), exprs).map(Lit::Int),
144        Expr::FlatIneq(_, a, b, c) => {
145            let a: i32 = a.try_into().ok()?;
146            let b: i32 = b.try_into().ok()?;
147            let c: i32 = c.try_into().ok()?;
148
149            Some(Lit::Bool(a <= b + c))
150        }
151        Expr::FlatSumGeq(_, exprs, a) => {
152            let sum = exprs.iter().try_fold(0, |acc, atom: &Atom| {
153                let n: i32 = atom.try_into().ok()?;
154                let acc = acc + n;
155                Some(acc)
156            })?;
157
158            Some(Lit::Bool(sum >= a.try_into().ok()?))
159        }
160        Expr::FlatSumLeq(_, exprs, a) => {
161            let sum = exprs.iter().try_fold(0, |acc, atom: &Atom| {
162                let n: i32 = atom.try_into().ok()?;
163                let acc = acc + n;
164                Some(acc)
165            })?;
166
167            Some(Lit::Bool(sum >= a.try_into().ok()?))
168        }
169        Expr::Min(_, e) => {
170            opt_vec_lit_op::<i32, i32>(|e| e.iter().min().copied(), e.as_ref()).map(Lit::Int)
171        }
172        Expr::Max(_, e) => {
173            opt_vec_lit_op::<i32, i32>(|e| e.iter().max().copied(), e.as_ref()).map(Lit::Int)
174        }
175        Expr::UnsafeDiv(_, a, b) | Expr::SafeDiv(_, a, b) => {
176            if unwrap_expr::<i32>(b)? == 0 {
177                return None;
178            }
179            bin_op::<i32, i32>(|a, b| ((a as f32) / (b as f32)).floor() as i32, a, b).map(Lit::Int)
180        }
181        Expr::UnsafeMod(_, a, b) | Expr::SafeMod(_, a, b) => {
182            if unwrap_expr::<i32>(b)? == 0 {
183                return None;
184            }
185            bin_op::<i32, i32>(|a, b| a - b * (a as f32 / b as f32).floor() as i32, a, b)
186                .map(Lit::Int)
187        }
188        Expr::MinionDivEqUndefZero(_, a, b, c) => {
189            // div always rounds down
190            let a: i32 = a.try_into().ok()?;
191            let b: i32 = b.try_into().ok()?;
192            let c: i32 = c.try_into().ok()?;
193
194            if b == 0 {
195                return None;
196            }
197
198            let a = a as f32;
199            let b = b as f32;
200            let div: i32 = (a / b).floor() as i32;
201            Some(Lit::Bool(div == c))
202        }
203        Expr::Bubble(_, a, b) => bin_op::<bool, bool>(|a, b| a && b, a, b).map(Lit::Bool),
204        Expr::MinionReify(_, a, b) => {
205            let result = eval_constant(a)?;
206
207            let result: bool = result.try_into().ok()?;
208            let b: bool = b.try_into().ok()?;
209
210            Some(Lit::Bool(b == result))
211        }
212        Expr::MinionReifyImply(_, a, b) => {
213            let result = eval_constant(a)?;
214
215            let result: bool = result.try_into().ok()?;
216            let b: bool = b.try_into().ok()?;
217
218            if b {
219                Some(Lit::Bool(result))
220            } else {
221                Some(Lit::Bool(true))
222            }
223        }
224        Expr::MinionModuloEqUndefZero(_, a, b, c) => {
225            // From Savile Row. Same semantics as division.
226            //
227            //   a - (b * floor(a/b))
228            //
229            // We don't use % as it has the same semantics as /. We don't use / as we want to round
230            // down instead, not towards zero.
231
232            let a: i32 = a.try_into().ok()?;
233            let b: i32 = b.try_into().ok()?;
234            let c: i32 = c.try_into().ok()?;
235
236            if b == 0 {
237                return None;
238            }
239
240            let modulo = a - b * (a as f32 / b as f32).floor() as i32;
241            Some(Lit::Bool(modulo == c))
242        }
243        Expr::MinionPow(_, a, b, c) => {
244            // only available for positive a b c
245
246            let a: i32 = a.try_into().ok()?;
247            let b: i32 = b.try_into().ok()?;
248            let c: i32 = c.try_into().ok()?;
249
250            if a <= 0 {
251                return None;
252            }
253
254            if b <= 0 {
255                return None;
256            }
257
258            if c <= 0 {
259                return None;
260            }
261
262            Some(Lit::Bool(a ^ b == c))
263        }
264        Expr::MinionWInIntervalSet(_, _, _) => None,
265        Expr::AllDiff(_, e) => {
266            let es = e.clone().unwrap_list()?;
267            let mut lits: HashSet<Lit> = HashSet::new();
268            for expr in es {
269                let Expr::Atomic(_, Atom::Literal(x)) = expr else {
270                    return None;
271                };
272                match x {
273                    Lit::Int(_) | Lit::Bool(_) => {
274                        if lits.contains(&x) {
275                            return Some(Lit::Bool(false));
276                        } else {
277                            lits.insert(x.clone());
278                        }
279                    }
280                    Lit::AbstractLiteral(_) => return None, // Reject AbstractLiteral cases
281                }
282            }
283            Some(Lit::Bool(true))
284        }
285        Expr::FlatAllDiff(_, es) => {
286            let mut lits: HashSet<Lit> = HashSet::new();
287            for atom in es {
288                let Atom::Literal(x) = atom else {
289                    return None;
290                };
291
292                match x {
293                    Lit::Int(_) | Lit::Bool(_) => {
294                        if lits.contains(x) {
295                            return Some(Lit::Bool(false));
296                        } else {
297                            lits.insert(x.clone());
298                        }
299                    }
300                    Lit::AbstractLiteral(_) => return None, // Reject AbstractLiteral cases
301                }
302            }
303            Some(Lit::Bool(true))
304        }
305        Expr::FlatWatchedLiteral(_, _, _) => None,
306        Expr::AuxDeclaration(_, _, _) => None,
307        Expr::Neg(_, a) => {
308            let a: &Atom = a.try_into().ok()?;
309            let a: i32 = a.try_into().ok()?;
310            Some(Lit::Int(-a))
311        }
312        Expr::Minus(_, a, b) => {
313            let a: &Atom = a.try_into().ok()?;
314            let a: i32 = a.try_into().ok()?;
315
316            let b: &Atom = b.try_into().ok()?;
317            let b: i32 = b.try_into().ok()?;
318
319            Some(Lit::Int(a - b))
320        }
321        Expr::FlatMinusEq(_, a, b) => {
322            let a: i32 = a.try_into().ok()?;
323            let b: i32 = b.try_into().ok()?;
324            Some(Lit::Bool(a == -b))
325        }
326        Expr::FlatProductEq(_, a, b, c) => {
327            let a: i32 = a.try_into().ok()?;
328            let b: i32 = b.try_into().ok()?;
329            let c: i32 = c.try_into().ok()?;
330            Some(Lit::Bool(a * b == c))
331        }
332        Expr::FlatWeightedSumLeq(_, cs, vs, total) => {
333            let cs: Vec<i32> = cs
334                .iter()
335                .map(|x| TryInto::<i32>::try_into(x).ok())
336                .collect::<Option<Vec<i32>>>()?;
337            let vs: Vec<i32> = vs
338                .iter()
339                .map(|x| TryInto::<i32>::try_into(x).ok())
340                .collect::<Option<Vec<i32>>>()?;
341            let total: i32 = total.try_into().ok()?;
342
343            let sum: i32 = izip!(cs, vs).fold(0, |acc, (c, v)| acc + (c * v));
344
345            Some(Lit::Bool(sum <= total))
346        }
347        Expr::FlatWeightedSumGeq(_, cs, vs, total) => {
348            let cs: Vec<i32> = cs
349                .iter()
350                .map(|x| TryInto::<i32>::try_into(x).ok())
351                .collect::<Option<Vec<i32>>>()?;
352            let vs: Vec<i32> = vs
353                .iter()
354                .map(|x| TryInto::<i32>::try_into(x).ok())
355                .collect::<Option<Vec<i32>>>()?;
356            let total: i32 = total.try_into().ok()?;
357
358            let sum: i32 = izip!(cs, vs).fold(0, |acc, (c, v)| acc + (c * v));
359
360            Some(Lit::Bool(sum >= total))
361        }
362        Expr::FlatAbsEq(_, x, y) => {
363            let x: i32 = x.try_into().ok()?;
364            let y: i32 = y.try_into().ok()?;
365
366            Some(Lit::Bool(x == y.abs()))
367        }
368        Expr::UnsafePow(_, a, b) | Expr::SafePow(_, a, b) => {
369            let a: &Atom = a.try_into().ok()?;
370            let a: i32 = a.try_into().ok()?;
371
372            let b: &Atom = b.try_into().ok()?;
373            let b: i32 = b.try_into().ok()?;
374
375            if (a != 0 || b != 0) && b >= 0 {
376                Some(Lit::Int(a.pow(b as u32)))
377            } else {
378                None
379            }
380        }
381        Expr::Scope(_, _) => None,
382        Expr::MinionElementOne(_, _, _, _) => None,
383    }
384}

386/// Evaluate the root expression.
387///
388/// This returns either Expr::Root([true]) or Expr::Root([false]).
389#[register_rule(("Constant", 9001))]
390fn eval_root(expr: &Expr, _: &SymbolTable) -> ApplicationResult {
391    // this is its own rule not part of apply_eval_constant, because root should return a new root
392    // with a literal inside it, not just a literal
393
394    let Expr::Root(_, exprs) = expr else {
395        return Err(RuleNotApplicable);
396    };
397
398    match exprs.len() {
399        0 => Ok(Reduction::pure(Expr::Root(
400            Metadata::new(),
401            vec![true.into()],
402        ))),
403        1 => Err(RuleNotApplicable),
404        _ => {
405            let lit =
406                vec_op::<bool, bool>(|e| e.iter().all(|&e| e), exprs).ok_or(RuleNotApplicable)?;
407
408            Ok(Reduction::pure(Expr::Root(
409                Metadata::new(),
410                vec![lit.into()],
411            )))
412        }
413    }
414}

480fn flat_op<T, A>(f: fn(Vec<T>, T) -> A, a: &[Expr], b: &Expr) -> Option<A>
481where
482    T: TryFrom<Lit>,
483{
484    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;

489fn unwrap_expr<T: TryFrom<Lit>>(expr: &Expr) -> Option<T> {
490    let c = eval_constant(expr)?;
491    TryInto::<T>::try_into(c).ok()
492}
